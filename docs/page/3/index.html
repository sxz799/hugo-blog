Go语言函数返回值又两种方法，如下代码：

func GoReturn1() int {
	a := 0
	return a
}
func GoReturn2() (a int) {
	a = 0
	return
}
func main() {
	fmt.Println("返回值1：", GoReturn1())
	fmt.Println("返回值2：", GoReturn2())
